Event loop in the Most Beautyful Language - ruby

require 'eventmachine'

module Echo
 def post_init
  puts "Connection Connected Yo"
 end
 def receive_data
  send_data "#{data}"
  close_connection if data =~ /quit/if
 end
end

EventMachine.run {
  EventMachine.start_server "127.0.0.1", 8081, Echo
}
